Pseudo-code: Banking Portal
BEGIN
    DEFINE CLASS Account
        ATTRIBUTE accountNumber
        ATTRIBUTE balance

        DEFINE METHOD withdraw(amount)
            // To be overridden by subclasses

        DEFINE METHOD deposit(amount)
            INCREASE balance BY amount

    DEFINE CLASS Saving EXTENDS Account
        ATTRIBUTE minimumBalance

        DEFINE METHOD withdraw(amount)
            IF balance - amount >= minimumBalance
                DECREASE balance BY amount
            ELSE
                PRINT "Insufficient funds"

    DEFINE CLASS Current EXTENDS Account
        ATTRIBUTE overdraftLimit

        DEFINE METHOD withdraw(amount)
            IF balance - amount >= -overdraftLimit
                DECREASE balance BY amount
            ELSE
                PRINT "Overdraft limit exceeded"

    DEFINE MAIN METHOD
        INITIALIZE savingAccount
        INITIALIZE currentAccount
        PERFORM deposit and withdraw operations ON accounts
END

Pseudo-code: Vehicle Rental System
BEGIN
    DEFINE ABSTRACT CLASS Vehicle
        ATTRIBUTE model
        ATTRIBUTE registrationNumber
        DEFINE ABSTRACT METHOD displayInfo()

    DEFINE INTERFACE Rentable
        DEFINE METHOD rent()

    DEFINE CLASS Car EXTENDS Vehicle IMPLEMENTS Rentable
        DEFINE METHOD displayInfo()
            PRINT "Car details"
        DEFINE METHOD rent()
            PRINT "Car rented"

    DEFINE CLASS Bike EXTENDS Vehicle IMPLEMENTS Rentable
        DEFINE METHOD displayInfo()
            PRINT "Bike details"
        DEFINE METHOD rent()
            PRINT "Bike rented"

    DEFINE CLASS VehicleFactory
        DEFINE METHOD createVehicle(type)
            IF type == "Car"
                RETURN new Car()
            ELSE IF type == "Bike"
                RETURN new Bike()

    DEFINE MAIN METHOD
        INITIALIZE vehicleList
        FOR EACH type IN requestedVehicleTypes
            vehicle = VehicleFactory.createVehicle(type)
            CALL vehicle.rent()
            CALL vehicle.displayInfo()
END
